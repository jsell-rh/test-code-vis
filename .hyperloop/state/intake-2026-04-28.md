import json, pathlib, subprocess

def load_resolved_set(state_dir=".hyperloop/state"):
    path = pathlib.Path(state_dir) / "resolved-specs.json"
    data = json.loads(path.read_text())
    return {entry["spec"] for entry in data["resolved"]}

def reopen_condition_triggered(spec):
    result = subprocess.run(

        capture_output=True, text=True

resolved = load_resolved_set()
